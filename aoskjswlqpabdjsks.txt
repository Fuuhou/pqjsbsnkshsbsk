def commit_and_push_to_github(file_path, commit_message):
    try:
        repo_path = os.path.dirname(file_path)
        os.system(f'cd {repo_path} && git add {file_path}')
        os.system(f'cd {repo_path} && git commit -m "{commit_message}"')
        os.system(f'cd {repo_path} && git push')

        logging.info(f"File {file_path} successfully pushed to GitHub with commit message: {commit_message}")
    except Exception as e:
        logging.error(f"Error during GitHub push: {str(e)}")

def load_host():
    try:
        # Memuat data host dari file 'host.json' di GitHub
        contents = repo.get_contents("host.json")
        return json.loads(contents.decoded_content.decode())
    except Exception as e:
        logging.error(f"Error loading host data from GitHub (host.json): {e}")
        return {}

def save_host(host_data, commit_message="Update host data"):
    try:
        # Memuat isi file 'host.json' dari GitHub
        contents = repo.get_contents("host.json")
        updated_content = json.dumps(host_data, indent=4)
        
        # Memperbarui file 'host.json' di GitHub
        repo.update_file(contents.path, commit_message, updated_content, contents.sha)
        logging.info("Host data successfully updated on GitHub.")
    except Exception as e:
        logging.error(f"Error updating host data on GitHub (host.json): {e}")

def execute_command(command: List[str]) -> str:
    """Execute a command on the server and return its output."""
    try:
        logging.info(f"Executing command: {command}")
        result = subprocess.run(command, capture_output=True, text=True, timeout=60)
        logging.info(f"Command output: {result.stdout}")
        if result.stderr:
            logging.error(f"Command error output: {result.stderr}")
        return result.stdout
    except subprocess.TimeoutExpired:
        logging.error(f"Command `{command}` timed out.")
        return "Command timed out."
    except OSError as e:
        logging.error(f"Error executing command `{command}`: {e}")
        return f"Error: {e}"

def check_spam(update: Update, context: CallbackContext):
    user_id = update.message.from_user.id
    current_time = time.time()

    # Append the current timestamp to the user's message timestamps
    user_message_timestamps[user_id].append(current_time)

    # Remove timestamps older than 30 seconds
    while user_message_timestamps[user_id] and user_message_timestamps[user_id][0] < current_time - 30:
        user_message_timestamps[user_id].popleft()

    # Check if the user has sent 100 or more messages in the last 30 seconds
    if len(user_message_timestamps[user_id]) >= 100:
        context.bot.restrict_chat_member(update.message.chat_id, user_id, until_date=current_time + 18000)  # Block for 5 hours
        update.message.reply_text(f"User `{user_id}` diblokir karena spamming. Auto-unblock dalam 5 jam.", parse_mode="Markdown")
        user_message_timestamps[user_id] = deque()  # Reset the message timestamps after blocking

        # Save blocked user data to blockspam.json
        blocked_user_data = {"user_id": user_id, "block_time": current_time}
        with open(BLOCK_SPAM, "a") as file:
            json.dump(blocked_user_data, file)
            file.write("\n")

        commit_and_push_to_github(BLOCK_SPAM, f"Block user `{user_id}` for spamming")

        # Send notification to Telegram channel
        message = f"User `{user_id}` diblokir karena spamming."
        context.bot.send_message(CHAT_ID, message, parse_mode="Markdown")

def send_notif(chat_id, message, parse_mode='Markdown'):
    payload = {
        'chat_id': chat_id,
        'text': message,
        'parse_mode': parse_mode
    }
    try:
        response = requests.post(LOGGING_API_URL, json=payload)
        response.raise_for_status()
    except requests.exceptions.RequestException as e:
        logging.error(f"Failed to send notification: {e}")

        logger.error(f"Kesalahan saat mengambil alamat IP server: {e}")

def get_provider_name(ip_address: str) -> str:
    try:
        response = requests.get(f'https://ipinfo.io/{ip_address}/org')
        response.raise_for_status()  # Raise an HTTPError for bad responses
        provider_info = response.text.strip()

        # Split the response by spaces and skip the first word
        parts = provider_info.split()
        if len(parts) > 1:
            return ' '.join(parts[1:])  # Join all words except the first one
        return provider_info  # Return as is if there's only one word
    except requests.RequestException as e:
        logging.error(f'Gagal mendapatkan nama provider: {str(e)}')
        return 'Tidak diketahui'

def get_location_info(ip_address: str) -> str:
    try:
        # Fetch country, timezone, and city using ipinfo.io
        country = requests.get(f'https://ipinfo.io/{ip_address}/country').text.strip()

        return f"{country} • {timezone} • {city}"
    except Exception as e:
        logging.error(f'Gagal mengambil informasi lokasi untuk {ip_address}: {str(e)}')
        return "Informasi lokasi tidak tersedia"

def ping_ip(ip_address: str) -> str:
    """Ping the given IP address and return the status as ON or OFF."""
    try:
        # Execute the ping command, the '-c 1' flag is for sending one packet
        output = subprocess.run(["ping", "-c", "1", ip_address], stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        
        # If the return code is 0, the ping was successful
        if output.returncode == 0:
            return "🟢 ON"
        else:
            return "🔴 OFF"
    except Exception as e:
        logging.error(f'Error pinging {ip_address}: {str(e)}')
        return "🔴 OFF"

def generate_unique_code(length=10):
    """Generate a unique gift code consisting of letters and digits."""
    return ''.join(random.choices(string.ascii_uppercase + string.ascii_lowercase + string.digits, k=length))

def redeem_gift_code(user_id, gift_code, user_fullname):
    """Redeem a gift code and update user coins."""
    gift_codes_data = load_data_from_github(GIFT_CODES_FILE)
    user_koin_data = load_data_from_github(FILE_PATH)

    if gift_code not in gift_codes_data:
        return "Kode gift tidak ada.", 0  # Return 0 coins if code is invalid

    # Initialize redeems if it doesn't exist
    if 'redeems' not in gift_codes_data[gift_code]:
        gift_codes_data[gift_code]['redeems'] = []

    # Check if user has already redeemed this gift code
    if user_id in {redeem['user_id'] for redeem in gift_codes_data[gift_code]['redeems']}:
        return "Kamu sudah menukarkan kode gift ini.", 0

    remaining_coins = gift_codes_data[gift_code]['coins']
    if remaining_coins <= 0:
        return "Tidak ada koin tersisa dalam kode gift ini.", 0

    # Randomly give coins to the user
    coins_to_give = random.randint(1, remaining_coins)
    user_koin_data.setdefault(str(user_id), {'coins': 0, 'join_date': datetime.now().strftime('%d-%m-%Y')})
    user_koin_data[str(user_id)]['coins'] += coins_to_give

    # Update the gift code data
    gift_codes_data[gift_code]['coins'] -= coins_to_give

    # Log the redemption
    redeem_info = {
        "user_id": user_id,
        "fullname": user_fullname,
        "date": datetime.now().strftime('%d-%m-%Y %H:%M')
    }
    gift_codes_data[gift_code]['redeems'].append(redeem_info)

    # Save updated data
    save_data_to_github(GIFT_CODES_FILE, gift_codes_data, "Redeem gift code")
    save_data_to_github(FILE_PATH, user_koin_data, "Update user koin after redeeming gift code")

    return (f"━━━━━━━━━━━━━━━━━━\n"
            f"*✰ REDEEM GIFT ✰*\n"
            f"━━━━━━━━━━━━━━━━━━\n"
            f"*Kamu Mendapat :* `{coins_to_give} Koin`\n"
            f"*Tanggal :* `{datetime.now().strftime('%d-%m-%Y %H:%M')} WIB`\n"
            f"━━━━━━━━━━━━━━━━━━"), coins_to_give

@registered_user_only
def gift(update: Update, context: CallbackContext):
    """Handle the /gift command to create and store a gift code with details or check gift details."""
    
    if len(context.args) == 0:
        update.message.reply_text("Gunakan perintah dengan format: /gift <koin> <hari> atau /gift cek <kode>")
        return

    if context.args[0] == "cek" and len(context.args) == 2:
        gift_cek(update, context)
        return

    # Logic for creating a gift code
    if len(context.args) != 2:
        update.message.reply_text("Gunakan perintah dengan format: /gift <koin> <hari>")
        return

    coins = context.args[0]
    days_active = context.args[1]

    if not coins.isdigit() or not days_active.isdigit():
        update.message.reply_text("Jumlah koin dan hari aktif harus berupa angka.")
        return

    user_id = update.effective_user.id
    fullname = update.effective_user.full_name
    code = generate_unique_code()

    # Calculate expiration time
    expiration_time = datetime.now() + timedelta(days=int(days_active))
    expiration_date = expiration_time.strftime("%d-%m-%Y %H:%M")

    # Check and deduct user's coins
    user_koin_data = load_data_from_github(FILE_PATH)
    user_koin_data.setdefault(str(user_id), {'coins': 0, 'join_date': datetime.now().strftime('%d-%m-%Y')})

    if user_koin_data[str(user_id)]['coins'] < int(coins):
        update.message.reply_text("Koin tidak cukup untuk membuat gift code.")
        return

    user_koin_data[str(user_id)]['coins'] -= int(coins)

    # Add gift code details
    gift_details = {
        "coins": int(coins),
        "expiration_time": expiration_date,
        "created_by": user_id,
        "created_by_fullname": fullname,
        "redeems": []
    }
    add_gift_code(code, gift_details)

    save_data_to_github(FILE_PATH, user_koin_data, "Update user koin after creating gift code")

    record_and_notify(user_id, f"User membuat kode Gift, isi {coins} Koin.")
    update.message.reply_text(
        f"━━━━━━━━━━━━━━━━━━\n"
        f"*✰ SHARE GIFTS ✰*\n"
        f"━━━━━━━━━━━━━━━━━━\n"
        f"*Coins :* `{coins}`\n"
        f"*Gift Code :* `{code}`\n"
        f"*By :* [{fullname}](tg://user?id={user_id})\n"
        f"*Expired :* `{expiration_date} WIB`\n"
        f"━━━━━━━━━━━━━━━━━━\n"
        f"🤖 *@superzoybot*", parse_mode="Markdown"
    )

def redeem(update: Update, context: CallbackContext):
    """Handle the /redeem command to redeem a gift code."""
    message = update.message.text.split()
    
    if len(message) != 2:
        update.message.reply_text("Gunakan perintah dengan format: /redeem <kode>")
        return

    code = message[1]
    user_id = update.effective_user.id
    user_fullname = update.effective_user.full_name
    result_message, coins_received = redeem_gift_code(user_id, code, user_fullname)

    # Notify the record with the coins received
    record_and_notify(user_id, f"User meredeem kode gift, dapat {coins_received} Koin.")
    update.message.reply_text(result_message, parse_mode="Markdown")

@registered_user_only
def gift_cek(update: Update, context: CallbackContext):
    """Check the gift code details."""
    
    message = update.message.text.split()
    if len(message) != 3 or message[1] != "cek":
        update.message.reply_text("Gunakan perintah dengan format: /gift cek <code>")

        return

    redeem_info = "\n".join(
        [f"➮ [{redeem['fullname']}](tg://user?id={redeem['user_id']})" for redeem in redeem_list] or
        ["Tidak ada yang meredeem."]
    )

    update.message.reply_text(
        f"━━━━━━━━━━━━━━━━━━\n"
        f"*Gift Code :* `{code}`\n"
        f"*Coins Left :* `{coins}`\n"
        f"*Expired :* `{expiration_time} WIB`\n"
        f"*Redeemed By:*\n{redeem_info}\n"
        f"━━━━━━━━━━━━━━━━━━\n"
        f"🤖 *@superzoybot*", parse_mode="Markdown"
    )

@registered_user_only
def add_host(update: Update, context: CallbackContext):
    try:
        message = update.message.text.split()
        if len(message) < 3:
            update.message.reply_text('Format perintah salah. Gunakan: /add_host <nama> <host>')
            return
        
        # Ambil nama dan host dari perintah
        nama_host = message[1]
        host_url = message[2]

        # Memuat data host dari GitHub
        host_data = load_host()

        # Tambahkan host baru
        host_data[nama_host] = host_url

        # Simpan data host yang diperbarui
        save_host(host_data, commit_message=f"Menambahkan host: {nama_host}")

        update.message.reply_text(f'Host {nama_host} berhasil ditambahkan.')
    except Exception as e:
        update.message.reply_text(f'Error menambahkan host: {str(e)}')

@admin_only
def del_host(update: Update, context: CallbackContext):
    try:
        message = update.message.text.split()
        if len(message) < 2:
            update.message.reply_text('Format perintah salah. Gunakan: /del_host <nama>')
            return
        
        # Ambil nama host yang akan dihapus
        nama_host = message[1]

        # Memuat data host dari GitHub
        host_data = load_host()

        # Hapus host jika ditemukan
        if nama_host in host_data:
            del host_data[nama_host]
            save_host(host_data, commit_message=f"Menghapus host: {nama_host}")
            update.message.reply_text(f'Host {nama_host} berhasil dihapus.')
        else:
            update.message.reply_text(f'Host {nama_host} tidak ditemukan.')

    except Exception as e:
        update.message.reply_text(f'Error menghapus host: {str(e)}')

        # Iterasi melalui data host dan tambahkan ke pesan

        update.message.reply_text(message, parse_mode='HTML')

    except Exception as e:
        update.message.reply_text(f'Error saat memuat daftar host: {str(e)}')

def build_curl_command(vpn_type, username, exp_days, host_url):
    """Membangun perintah curl dengan jq untuk memproses output JSON."""
    return f"""curl -sS -X POST {host_url}renew-{vpn_type} \\
    -H "X-API-KEY: slime123" \\
    -H "Content-Type: application/json" \\
    -d '{{"user":"{username}", "exp":"{exp_days}"}}' | jq -r '.stdout'"""

def run_curl_command(vpn_type, username, exp_days, host_url):
    """Menjalankan perintah curl dan mengembalikan hasilnya."""
    command = build_curl_command(vpn_type, username, exp_days, host_url)
    try:
        result = subprocess.run(command, capture_output=True, text=True, shell=True, check=True)
        return result.stdout, result.stderr
    except subprocess.CalledProcessError as e:
        return e.stdout, e.stderr

def format_output(vpn_type, output, provider_name, username):
    """Memformat output sesuai dengan jenis VPN dan teks yang diinginkan."""
    formatted_output = "━━━━━━━━━━━━━━━━━━━━\n"

    if vpn_type == "ssh":
        formatted_output += "*Successfully Renewed SSH*\n"
        expiration_date = None
        formatted_output += "━━━━━━━━━━━━━━━━━━━━\n"

        for line in output.splitlines():
            if "Username" in line:
                username = line.split(':')[1].strip()
            elif "Expires on" in line:
                expiration_date = line.split(':')[1].strip()

        if username and expiration_date:
            formatted_output += f"*Provider :* `{provider_name}`\n"
            formatted_output += f"*Username :* `{username}`\n"
            formatted_output += f"*Expired On :* `{expiration_date}`\n"
            formatted_output += "━━━━━━━━━━━━━━━━━━━━\n"

    else:
        date_pattern = re.compile(r"\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2} WIB")

        for line in output.splitlines():
            if "Successfully Renewed" in line:
                formatted_output += f"*Successfully Renewed {vpn_type.upper()}*\n"
            elif "Client Name :" in line:
                formatted_output += f"*Provider :* `{provider_name}`\n"
                formatted_output += f"*Client Name :* `{line.split(':')[1].strip()}`\n"
            elif "Expired On  :" in line:
                match = date_pattern.search(line)
                if match:
                    formatted_output += f"*Expired On :* `{match.group()}`\n"
            else:

def is_vpn_expired(user_id, username, vpn_type, expired_data):
    for entry in expired_data.get(user_id, []):
        if entry["username"] == username and (vpn_type == entry.get("vpn_type") or vpn_type == "ssh"):
            expiration_date = datetime.strptime(entry["expired"], '%d-%m-%Y %H:%M')
            return expiration_date < datetime.now()  # True if expired
    return False  # Not found, not expired

@registered_user_only
def renew_vpn(update, context):
    try:
        # Load necessary data
        user_id = str(update.message.from_user.id)
        waiting_message = update.message.reply_text("⏳")

        # Load all user data (not just the current user)
        all_user_data = load_data_from_github(FILE_PATH)
        user_data = all_user_data.get(user_id, {})
        expired_data = load_data_from_github(AKUN_VPN_PATH)
        host_data = load_data_from_github(HOST_FILE_PATH)

        if not host_data:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Gagal memuat URL host. Periksa file konfigurasi.')
            return

        # Check command arguments
        if len(context.args) != 3:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Format perintah salah. Gunakan /renew <tipe_vpn> <username> <coins>.')
            return

        vpn_type = context.args[0].lower()
        username = context.args[1]
        pay_coins = int(context.args[2])

        # Check maximum coins limit
        if pay_coins > 50:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Jumlah koin maksimal adalah 50.')
            return

        # Validate VPN type
        valid_vpn_types = ["ssh", "vmessws", "vlessws", "trojanws", "vmessgrpc", "vlessgrpc", "trojangrpc"]
        if vpn_type not in valid_vpn_types:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text(f"Jenis VPN tidak valid. Gunakan salah satu dari: {', '.join(valid_vpn_types)}.", parse_mode="Markdown")
            return

        # Check if user has enough coins
        if user_data.get('coins', 0) < pay_coins:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Koin kamu tidak cukup.')
            return

        # Deduct coins from user
        user_data['coins'] -= pay_coins
        user_data['coins_usage'] = user_data.get('coins_usage', 0) + pay_coins

        # Save updated user data back to all users' data
        all_user_data[user_id] = user_data
        save_data_to_github(FILE_PATH, all_user_data, "Update user koin")

        # Check if VPN is expired for this user
        vpn_entry = next((entry for entry in expired_data.get(user_id, []) if entry["username"] == username), None)
        if not vpn_entry:
            rollback_coins(user_data, pay_coins, all_user_data)  # Rollback coins if VPN entry is not found
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text(f'Username `{username}` tidak ditemukan untuk pengguna ini. Transaksi dibatalkan dan koin dikembalikan.', parse_mode="Markdown")
            return

        label = vpn_entry.get("label")
        server_name = vpn_entry.get("host")

        # Get price based on server and label
        price = get_price_for_server_and_label(server_name, label)
        if price == 0:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text(f'Harga untuk server {server_name} dengan label {label} tidak ditemukan.')
            return

        # Calculate new expiration days
        if vpn_type == "ssh":
            remaining_days = (datetime.strptime(vpn_entry["expired"], '%d-%m-%Y %H:%M') - datetime.now()).days
            exp_days = remaining_days + (pay_coins * price) + 1
        else:
            exp_days = pay_coins * price  # VPN types other than SSH

        # Update expiration date
        current_expiration = datetime.strptime(vpn_entry["expired"], '%d-%m-%Y %H:%M')
        new_expiration_date = current_expiration + timedelta(days=exp_days)
        vpn_entry["expired"] = new_expiration_date.strftime('%d-%m-%Y %H:%M')

        # Save updated VPN expiration data
        save_data_to_github(AKUN_VPN_PATH, expired_data, "Update VPN expiration")

        # Determine host URL
        host_url = host_data.get(server_name)
        if not host_url:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text(f'Host URL untuk {server_name} tidak ditemukan dalam data konfigurasi.')
            return

        # Run curl command
        output, error = run_curl_command(vpn_type, username, exp_days, host_url)

        # Process output from curl command
        if error:
            rollback_coins(user_data, pay_coins, all_user_data)
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text(f'Terjadi kesalahan saat memperbarui *{vpn_type.upper()}* VPN. Transaksi dibatalkan dan koin dikembalikan.', parse_mode="Markdown")
            return

        # Check specific error
        if "IP Address Not Found In Our Database" in output:
            rollback_coins(user_data, pay_coins, all_user_data)
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Proses gagal karena IP Address tidak ditemukan dalam database kami. Koin kamu telah dikembalikan.', parse_mode="Markdown")
            return

        if "Script Expired !" in output:
            rollback_coins(user_data, pay_coins, all_user_data)
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Proses gagal karena script expired. Koin kamu telah dikembalikan.', parse_mode="Markdown")
            return

        if "Kedua URL gagal diakses!" in output:
            rollback_coins(user_data, pay_coins, all_user_data)
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Proses gagal karena URL gagal diakses. Koin kamu telah dikembalikan.', parse_mode="Markdown")
            return

        if "Not Found" in output:
            rollback_coins(user_data, pay_coins, all_user_data)
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Proses gagal karena URL gagal diakses. Koin kamu telah dikembalikan.', parse_mode="Markdown")
            return

        # Load provider information
        ip_data = load_data_from_github(SERVER_PATH)
        ip_address = ip_data.get(server_name)
        provider_name = get_provider_name(ip_address)

        # Format output
        formatted_output = format_output(vpn_type, output, provider_name, username)

        # Notify user and log transaction
        record_and_notify(user_id, f"User memperpanjang expired vpn, host: {server_name} - tipe: {vpn_type} - nama: {username} - koin: {pay_coins} - label: {label}")
        context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
        update.message.reply_text(formatted_output, parse_mode="Markdown")

    except Exception as e:
        logger.error(f"Terjadi kesalahan: {e}")
        # Rollback coins on general exception
        if user_id in all_user_data and all_user_data.get(user_id, {}).get('coins', 0) < pay_coins:
            rollback_coins(user_data, pay_coins, all_user_data)
        context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
        update.message.reply_text(f'Error: {e}')

@registered_user_only
def trial_vpn(update, context):
    user_id = str(update.message.from_user.id)
    waiting_message = update.message.reply_text("⏳")

    if len(context.args) < 2:
        context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
        update.message.reply_text('Format perintah salah. Gunakan format: /trial <server> <tipe_vpn>.')
        return

    host = context.args[0]
    vpn_type = context.args[1].lower()

    # Validate the VPN type
    valid_vpn_types = ["ssh", "vmessws", "vlessws", "trojanws", "vmessgrpc", "vlessgrpc", "trojangrpc"]
    if vpn_type not in valid_vpn_types:
        context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
        update.message.reply_text(f"Jenis VPN tidak valid. Pilih salah satu: `ssh`, `vmessws`, `vlessws`, `trojanws`, `vmessgrpc`, `vlessgrpc`, `trojangrpc`.", parse_mode="Markdown")
        return

    # Load the last trial data from GitHub
    last_trial_data = load_data_from_github(LAST_TRIAL_FILE)

    # Get today's date
    today = datetime.now().strftime('%d-%m-%Y')

    # Check if the user has tried today
    if user_id in last_trial_data and last_trial_data[user_id] == today:
        context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
        update.message.reply_text("Kamu sudah buat trial hari ini, coba lagi besok.")
        return

    try:
        # Load host URL
        host_data = load_data_from_github(HOST_FILE_PATH)
        host_url = host_data.get(host)
        ip_data = load_data_from_github(SERVER_PATH)
        ip_address = ip_data[host]
        provider_name = get_provider_name(ip_address)

        if not host_url:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text(f"Host `{host}` tidak ditemukan. Pastikan host sudah benar.")
            return

        # Run the curl command for the specified VPN type
        curl_command = f"""curl -sS -X POST {host_url}trial-{vpn_type} \\
            -H "X-API-KEY: slime123" \\
            -H "Content-Type: application/json" \\
            -d '{{"exp":"1", "limit_ip":"2"}}' | jq -r '.stdout'"""
        
        result = subprocess.run(curl_command, capture_output=True, text=True, shell=True)

        if result.returncode != 0:
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text(f"Error executing command: {result.stderr}")
            return

        # Get and clean the output
        filtered_output = result.stdout

        # Apply regex replacements for defined patterns
        for pattern, replacement in XRAY_PATTERNS.items():
            filtered_output = re.sub(pattern, replacement, filtered_output)

        # Replace "IP-Addres :" with "IP Address :"
        filtered_output = re.sub(r"IP-Addres :", "IP Address :", filtered_output)

        # Insert Provider information directly above "IP Address :"
        filtered_output = re.sub(r"(IP Address :|IP-Addres :)", f"Provider : {provider_name}\n\\1", filtered_output)

        # Clean output specific to SSH or any other type
        if vpn_type == "ssh":
            lines = filtered_output.splitlines()
            filtered_lines = [line for line in lines if 'Membuat Akun:' not in line and 'Setting Password:' not in line]
            filtered_output = '\n'.join(filtered_lines)

        # Check for errors in the output
        if "error" in filtered_output.lower():
            context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
            update.message.reply_text('Terdapat kesalahan saat membuat trial VPN. Silakan coba lagi.', parse_mode="Markdown")
            return

        # Store today's date as the last trial date for the user
        last_trial_data[user_id] = today
        save_data_to_github(LAST_TRIAL_FILE, last_trial_data, "Update last trial date")

        # Log trial VPN creation
        record_and_notify(user_id, f"User membuat trial vpn, host: {host} - tipe: {vpn_type}")

        # Send the result to the user
        context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
        update.message.reply_text(f"{filtered_output}\n🏬 <b>@xiestorez</b>", parse_mode="HTML")

    except Exception as e:
        logger.error(f'Error running trial command: {str(e)}')
        context.bot.delete_message(chat_id=update.effective_chat.id, message_id=waiting_message.message_id)
        update.message.reply_text(f"Gagal menjalankan perintah trial: `{str(e)}`", parse_mode="Markdown")

        update.message.reply_text("Format salah. Gunakan perintah: /transfer <user_id_tujuan> <jumlah_koin>")
        return

    target_user_id, coins = message[1], message[2]
    if not coins.isdigit():
        update.message.reply_text("Jumlah koin harus berupa angka.")
        return

    coins = int(coins)
    sender_id = str(update.effective_user.id)

    # Load data for coins and user information
    data = load_data_from_github(FILE_PATH)
    user_data = load_data_from_github(USER_DATA_PATH)

    # Check if sender exists in the data
    if sender_id not in data:
        update.message.reply_text("User ID pengirim tidak ditemukan.")
        return

    if sender_coins < coins:
        update.message.reply_text("Koin tidak cukup untuk melakukan transfer.")

    if target_user_id not in data:

    # Compose the transfer message in the requested format
    tf_msg = (
        f"━━━━━━━━━━━━━━━━━━\n"

        f"━━━━━━━━━━━━━━━━━━\n"
        f"*❃ Pengirim*\n"
        f"*⋗ Nama :* `{sender_full_name}`\n"
        f"*⋗ User ID :* [{sender_id}](tg://user?id={sender_id})\n"
        f"*⋗ Username :* @{sender_username}\n"
        f"━━\n"
        f"*❃ Penerima*\n"
        f"*⋗ Nama :* `{target_full_name}`\n"

@admin_only
def change_vpn_data(update: Update, context: CallbackContext) -> None:
    try:
        # Ensure three arguments are passed (operation, IP address, and value)
        if len(context.args) != 3:
            update.message.reply_text('Gunakan perintah dengan format: /change (limit/host) <ip_address> <limit_stok/nama_host>')
            return

        operation = context.args[0].lower()
        ip_address = context.args[1]
        new_value = context.args[2]

        # Load stock data
        stock_data = load_data_from_github(STOCK_PATH)

        # Check if the IP address exists in the stock file
        if ip_address not in stock_data:
            update.message.reply_text(f'IP address `{ip_address}` tidak ditemukan.', parse_mode="Markdown")
            return

        # Perform operation based on the type (limit or host)
        if operation == "limit":
            try:
                # Convert the new limit to an integer
                new_limit = int(new_value)

                # Set the new stock limit for the specified IP address
                stock_data[ip_address]["limit_stock_vpn"] = new_limit

                # Ensure current stock does not exceed the new limit
                stock_data[ip_address]["stock_vpn"] = min(stock_data[ip_address]["stock_vpn"], new_limit)

                update.message.reply_text(f'Limit stok VPN untuk IP `{ip_address}` berhasil diubah menjadi `{new_limit}`.', parse_mode="Markdown")
            except ValueError:
                update.message.reply_text('Batas stok harus berupa angka.', parse_mode="Markdown")
        
        elif operation == "host":
            # Update the host name for the specified IP address
            stock_data[ip_address]["host"] = new_value
            update.message.reply_text(f'Nama host untuk IP `{ip_address}` berhasil diubah menjadi `{new_value}`.', parse_mode="Markdown")
        
        else:
            update.message.reply_text('Operasi tidak valid. Gunakan "limit" atau "host".', parse_mode="Markdown")
            return

        # Save the updated stock data back to the file
        save_data_to_github(STOCK_PATH, stock_data, f"Change {operation} for {ip_address}")

    except Exception as e:
        logging.error(f'Gagal mengubah data VPN: {str(e)}')
        update.message.reply_text(f'Gagal mengubah data VPN: `{str(e)}`', parse_mode="Markdown")

def update_vpn_stock():
  try:
    # Load data from AKUN_VPN_PATH
    expired_data = load_data_from_github(AKUN_VPN_PATH)

    # Load current stock data
    stock_data = load_data_from_github(STOCK_PATH)

    # Initialize a dictionary to count hosts
    host_counts = defaultdict(int)

    # Count hosts in AKUN_VPN_PATH
    for user_info in expired_data.values():
      for entry in user_info:
        host = entry.get("host")
        if host:
          host_counts[host] += 1

    # Iterate through stock data to update stock_vpn
    for ip_address, stock_info in stock_data.items():
      host = stock_info.get("host")
      limit_stock_vpn = stock_info.get("limit_stock_vpn", 0)

      # Get the total count of expired hosts for the current stock host
      total_expired_count = host_counts.get(host, 0)

      # Calculate new stock
      new_stock_vpn = limit_stock_vpn - total_expired_count
      stock_info["stock_vpn"] = max(new_stock_vpn, 0) # Ensure stock doesn't go below 0

    # Save the updated stock data
    save_data_to_github(STOCK_PATH, stock_data, "Updated VPN stock based on host counts")

  except Exception as e:
    logging.error(f'Gagal memperbarui stok VPN: {str(e)}')

def send_message_in_parts(update: Update, message: str, chunk_size: int = 4096) -> None:
    """Sends a long message in parts if it exceeds chunk_size, using Markdown formatting."""
    
    # Split the message into chunks without worrying about HTML tags
    parts = [message[i:i + chunk_size] for i in range(0, len(message), chunk_size)]
    
    for part in parts:

def delete_expired_vpns():
    try:
        # Load expired data and hosts data
        expired_data = load_data_from_github(AKUN_VPN_PATH)
        hosts_data = load_data_from_github(HOST_FILE_PATH)

        # Get the current date
        current_date = datetime.now()

        # List to hold expired SSH VPNs to delete
        expired_ssh_vpns = []

        # Iterate over user VPN data
        for user_id, vpns in expired_data.items():
            for vpn in vpns:
                if vpn.get('vpn_type') == 'ssh':
                    expiration_date = datetime.strptime(vpn['expired'], '%d-%m-%Y %H:%M')

                    if current_date >= expiration_date:
                        # Append expired SSH VPNs to the list
                        expired_ssh_vpns.append(vpn)

        # Process each expired SSH VPN one by one
        for vpn in expired_ssh_vpns:
            result = delete_ssh_vpn(vpn, hosts_data)
            if result['success']:
                logger.info(f'Successfully deleted expired SSH VPN for user {vpn["username"]} on host {vpn["host"]}')
            else:
                logger.warning(f'Failed to delete expired SSH VPN for user {vpn["username"]}: {result["message"]}')

        # Save updated expired data without modification
        save_data_to_github(AKUN_VPN_PATH, expired_data, "Checked expired VPN data without deletion")

    except Exception as e:
        logger.error(f'Error while processing expired VPNs: {str(e)}')

def delete_ssh_vpn(vpn, hosts_data):
    """Helper function to delete an SSH VPN and return the result."""
    input_host = vpn['host']
    username = vpn['username']

    curl_command = f"""curl -sS -X DELETE {hosts_data[input_host]}del-ssh \\
    -H "Content-Type: application/json" \\
    -H "X-API-KEY: slime123" \\
    -d '{{"user": "{username}"}}' | jq -r '.stdout'"""

    result = subprocess.run(curl_command, capture_output=True, text=True, shell=True)

    if result.stdout.strip() and "user not found" not in result.stdout:
        return {'success': True}
    return {'success': False, 'message': result.stdout}

@admin_only
def delete_vpn(update: Update, context: CallbackContext):
    try:
        # Parse the command message
        message = update.message.text.split()
        user_id = str(update.message.from_user.id)

        # Check command format
        if len(message) < 3:
            update.message.reply_text(
                'Format perintah salah. Gunakan: /delete <tipe_vpn> <username>\n'
                'Contoh: /delete ssh myusername'
            )
            return

        vpn_type = message[1].lower()
        username = message[2]

        # Validate VPN type
        valid_vpn_types = ["ssh", "vmessws", "vlessws", "trojanws", "vmessgrpc", "vlessgrpc", "trojangrpc"]
        if vpn_type not in valid_vpn_types:
            update.message.reply_text(
                f"Tipe VPN tidak valid. Gunakan salah satu dari: {', '.join(valid_vpn_types)}.\n"
                f'Silakan cek tipe VPN yang tersedia dan coba lagi.', parse_mode="Markdown"
            )
            return

        # Load all VPN data from a single path
        vpn_data = load_data_from_github(AKUN_VPN_PATH)

        # Check if there is any VPN data at all
        if not vpn_data:
            update.message.reply_text('Tidak ada data VPN yang tersedia di sistem.', parse_mode="Markdown")
            return

        # Flag to check if any VPN entry was deleted
        vpn_deleted = False
        deleted_vpn_info = []

        # Loop through all users and search for the VPN entry to delete
        for user in vpn_data:
            user_vpn_data = vpn_data[user]
            vpn_entry = next((entry for entry in user_vpn_data if entry["username"] == username and entry["vpn_type"] == vpn_type), None)

            if vpn_entry:
                # Extract VPN details for confirmation
                vpn_username = vpn_entry.get("username", "Tidak diketahui")
                host = vpn_entry.get("host", "Tidak diketahui")
                expiration_date = vpn_entry.get("expired", "Tidak diketahui")

                # Store user_id (assuming `user` is the user_id in the vpn_data)
                vpn_user_id = user

                # Remove the VPN entry
                user_vpn_data.remove(vpn_entry)
                vpn_data[user] = user_vpn_data  # Update the user's VPN data

                # Track the deleted VPN info for feedback
                deleted_vpn_info.append(f"*User ID :* `{vpn_user_id}`\n*Tipe :* `{vpn_type}`\n*Username :* `{vpn_username}`\n*Server :* `{host}`\n*Expired :* `{expiration_date}`\n")
                vpn_deleted = True

        # If no VPN entry was deleted
        if not vpn_deleted:
            update.message.reply_text(f'VPN dengan username `{username}` dan tipe `{vpn_type}` tidak ditemukan di seluruh data.', parse_mode="Markdown")
            return

        # Save updated data back to the file on GitHub
        save_data_to_github(AKUN_VPN_PATH, vpn_data, "Hapus VPN entry dari semua pengguna")

        # Notify user of successful deletion with detailed information
        update.message.reply_text(
            f'━━━━━━━━━━━━━━━━━━\n'
            f'*Akun VPN Berhasil Dihapus*\n'
            f'_Informasi VPN :_\n'
            f'━━━━━━━━━━━━━━━━━━\n'
            + "\n".join(deleted_vpn_info) +
            f'━━━━━━━━━━━━━━━━━━', parse_mode="Markdown"
        )

    except Exception as e:
        logger.error(f'Gagal menghapus {vpn_type}: {str(e)}')
        update.message.reply_text(f'Gagal menghapus *{vpn_type.upper()}* dengan username `{username}` : `{str(e)}`', parse_mode="Markdown")

@registered_user_only
def set_notification_status(update: Update, context: CallbackContext):
    user_id = str(update.effective_user.id)
    command = context.args[0].lower() if context.args else None
    
    if command not in ['on', 'off']:
        update.message.reply_text("Perintah tidak valid. Gunakan '/notif on' atau '/notif off'.")
        return

    # Load notification settings directly from GitHub
    notif_settings = load_data_from_github(NOTIF_PATH)

    # Tentukan status baru berdasarkan perintah
    new_status = command == 'on'
    
    # Jika user_id belum ada, daftarkan data baru

        notif_settings[user_id]['enabled'] = new_status

    # Simpan perubahan ke GitHub
    save_data_to_github(NOTIF_PATH, notif_settings, "Update notification settings")

def remove_expired_vpn_accounts() -> None:
    """Remove expired VPN accounts."""
    try:
        expired_data = load_data_from_github(AKUN_VPN_PATH)
        current_time = datetime.now()

        # Remove expired VPN accounts
        for user_id, entries in list(expired_data.items()):
            valid_entries = [
                entry for entry in entries if datetime.strptime(entry["expired"], '%d-%m-%Y %H:%M') >= current_time
            ]
            if valid_entries:
                expired_data[user_id] = valid_entries
            else:
                del expired_data[user_id]

        # Save the updated data back
        save_data_to_github(AKUN_VPN_PATH, expired_data, "Cleanup expired VPN accounts")
        logger.info("Expired VPN accounts have been removed successfully.")

    except Exception as e:
        logger.error(f"Error removing expired VPN accounts: {e}")

def expired_notif(data: Dict[str, List[Dict[str, Any]]], current_date: datetime, notif_settings: Dict[str, Any]) -> None:
    for user_id, entries in data.items():
        for entry in entries:
            vpn_type = entry.get('vpn_type', 'Unknown VPN')  # Get vpn_type from entry
            try:
                expiration_date = datetime.strptime(entry['expired'], '%d-%m-%Y %H:%M')
            except ValueError as e:
                logger.error(f'Invalid date format for user {user_id}: {str(e)}')
                continue  # Skip invalid entries

            if expiration_date < current_date:  # Entry is expired
                message = (
                    f"━━━━━━━━━━━━━━━━━━\n"
                    f"🚫 *Akun {vpn_type.upper()} telah expired*\n"
                    f"*Username :* `{entry['username']}`\n"
                    f"*Server :* `{entry['host']}`\n"
                    f"*Expired On :* `{expiration_date.strftime('%d-%m-%Y %H:%M')} WIB`\n"
                    f"━━━━━━━━━━━━━━━━━━"
                )
                notify_user(user_id, message, notif_settings)
                logger.info(f"{vpn_type} notification sent for user {user_id}: {entry['username']}")

def notify_user(user_id: str, message: str, notif_settings: Dict[str, Any]) -> None:
    if notif_settings.get(user_id, {}).get('enabled', False):
        try:

            logger.info(f'Notified user {user_id}: {message}')
        except Exception as e:
            logger.error(f'Failed to send notification to user {user_id}: {str(e)}')
    else:
        logger.info(f'Notifications are disabled for user {user_id}. No message sent.')

def reminder_notif(data: Dict[str, List[Dict[str, Any]]], current_date: datetime, notif_settings: Dict[str, Any]) -> None:
    reminder_period_start = timedelta(days=1)  # 1 hari sebelum expired
    reminder_period_end = timedelta(days=3)    # 3 hari sebelum expired

    for user_id, entries in data.items():
        for entry in entries:
            vpn_type = entry.get('vpn_type', 'Unknown VPN')  # Get vpn_type from entry
            try:
                expiration_date = datetime.strptime(entry['expired'], '%d-%m-%Y %H:%M')
            except ValueError as e:
                logger.error(f'Invalid date format for user {user_id}: {str(e)}')
                continue  # Skip invalid entries

            # Calculate the range for the reminder
            days_until_expiration = expiration_date - current_date

            # If the expiration date is within 1 to 3 days from the current date
            if reminder_period_start <= days_until_expiration <= reminder_period_end:
                message = (
                    f"━━━━━━━━━━━━━━━━━━\n"
                    f"⚠️ *Akun {vpn_type.upper()} akan segera expired*\n"
                    f"*Username :* `{entry['username']}`\n"
                    f"*Server :* `{entry['host']}`\n"
                    f"*Expired On :* `{expiration_date.strftime('%d-%m-%Y %H:%M')} WIB`\n"
                    f"━━━━━━━━━━━━━━━━━━"
                )
                notify_user(user_id, message, notif_settings)
                logger.info(f"Reminder sent for {vpn_type} account of user {user_id}: {entry['username']}")

def reminder_expired_vpn_data() -> None:
